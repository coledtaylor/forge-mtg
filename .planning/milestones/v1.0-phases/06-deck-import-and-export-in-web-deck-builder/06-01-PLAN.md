<objective>
Create backend REST endpoints for deck import parsing and deck export formatting. The parse endpoint wraps Forge's existing DeckRecognizer to accept raw deck list text and return typed, color-coded parse results as JSON. The export endpoint loads a saved deck and formats it in one of four formats (generic text, MTGO, Arena, Forge .dck).

Purpose: All parsing and formatting logic lives server-side, leveraging DeckRecognizer's battle-tested multi-format parser and the card database for accurate card resolution. The frontend will consume these endpoints for live preview and export.
Output: Two new REST endpoints registered in WebServer, a new handler class, and a parse token DTO.
</objective>

    ```java
    package forge.web.dto;
    // imports: forge.deck.DeckRecognizer, forge.deck.DeckSection, forge.item.PaperCard

    public class ParseTokenDto {
        public String type;           // TokenType.name() e.g. "LEGAL_CARD", "UNKNOWN_CARD"
        public int quantity;           // from token.getQuantity()
        public String text;            // from token.getText()
        public String cardName;        // resolved PaperCard.getName() or null
        public String setCode;         // PaperCard.getEdition() or null
        public String collectorNumber; // PaperCard.getCollectorNumber() or null
        public String section;         // DeckSection.name() or null, treat null tokenSection as null (frontend defaults to Main)

        public static ParseTokenDto from(DeckRecognizer.Token token) {
            ParseTokenDto dto = new ParseTokenDto();
            dto.type = token.getType().name();
            dto.quantity = token.getQuantity();
            dto.text = token.getText();
            if (token.isCardToken() && token.getCard() != null) {
                PaperCard pc = token.getCard();
                dto.cardName = pc.getName();
                dto.setCode = pc.getEdition();
                dto.collectorNumber = pc.getCollectorNumber();
            }
            dto.section = token.getTokenSection() != null ? token.getTokenSection().name() : null;
            return dto;
        }
    }
    ```

    ```java
    package forge.web.api;
    // imports: io.javalin.http.Context, forge.deck.*, forge.item.PaperCard, forge.model.FModel,
    //          forge.localinstance.properties.ForgeConstants, forge.web.dto.ParseTokenDto,
    //          java.io.File, java.util.*, forge.deck.io.DeckSerializer

    public final class DeckImportExportHandler {
        private DeckImportExportHandler() { }

        // POST /api/decks/parse — body: { "text": "4 Lightning Bolt\n..." }
        public static void parse(final Context ctx) {
            @SuppressWarnings("unchecked")
            Map<String, String> body = ctx.bodyAsClass(Map.class);
            String text = body.get("text");
            if (text == null || text.trim().isEmpty()) {
                ctx.json(List.of());
                return;
            }
            String[] lines = text.split("\\r?\\n");
            DeckRecognizer recognizer = new DeckRecognizer();
            List<DeckRecognizer.Token> tokens = recognizer.parseCardList(lines);
            List<ParseTokenDto> result = new ArrayList<>();
            for (DeckRecognizer.Token token : tokens) {
                result.add(ParseTokenDto.from(token));
            }
            ctx.json(result);
        }

        // GET /api/decks/{name}/export?format=generic|mtgo|arena|forge
        public static void export(final Context ctx) {
            String name = ctx.pathParam("name");
            String format = ctx.queryParam("format");
            if (format == null) format = "generic";

            File deckFile = findDeckFile(name);
            if (deckFile == null) {
                ctx.status(404).json(Map.of("error", "Deck not found"));
                return;
            }
            Deck deck = DeckSerializer.fromFile(deckFile);
            if (deck == null) {
                ctx.status(404).json(Map.of("error", "Deck not found"));
                return;
            }

            String text = switch (format.toLowerCase()) {
                case "mtgo" -> formatMtgo(deck);
                case "arena" -> formatArena(deck);
                case "forge" -> formatForge(deck);
                default -> formatGeneric(deck);
            };
            ctx.json(Map.of("text", text));
        }

        // Generic: "4 Lightning Bolt"
        private static String formatGeneric(Deck deck) {
            StringBuilder sb = new StringBuilder();
            appendGenericSection(sb, deck.getOrCreate(DeckSection.Main));
            CardPool side = deck.get(DeckSection.Sideboard);
            if (side != null && !side.isEmpty()) {
                sb.append("\nSideboard\n");
                appendGenericSection(sb, side);
            }
            CardPool cmdr = deck.get(DeckSection.Commander);
            if (cmdr != null && !cmdr.isEmpty()) {
                sb.append("\nCommander\n");
                appendGenericSection(sb, cmdr);
            }
            return sb.toString().trim();
        }
        private static void appendGenericSection(StringBuilder sb, CardPool pool) {
            for (Map.Entry<PaperCard, Integer> e : pool) {
                sb.append(e.getValue()).append(" ").append(e.getKey().getName()).append("\n");
            }
        }

        // MTGO: "4 Lightning Bolt (2XM)"
        private static String formatMtgo(Deck deck) {
            StringBuilder sb = new StringBuilder();
            appendMtgoSection(sb, deck.getOrCreate(DeckSection.Main));
            CardPool side = deck.get(DeckSection.Sideboard);
            if (side != null && !side.isEmpty()) {
                sb.append("\nSideboard\n");
                appendMtgoSection(sb, side);
            }
            CardPool cmdr = deck.get(DeckSection.Commander);
            if (cmdr != null && !cmdr.isEmpty()) {
                sb.append("\nCommander\n");
                appendMtgoSection(sb, cmdr);
            }
            return sb.toString().trim();
        }
        private static void appendMtgoSection(StringBuilder sb, CardPool pool) {
            for (Map.Entry<PaperCard, Integer> e : pool) {
                PaperCard pc = e.getKey();
                sb.append(e.getValue()).append(" ").append(pc.getName())
                  .append(" (").append(pc.getEdition()).append(")\n");
            }
        }

        // Arena: "4 Lightning Bolt (2XM) 123"
        private static String formatArena(Deck deck) {
            StringBuilder sb = new StringBuilder();
            appendArenaSection(sb, deck.getOrCreate(DeckSection.Main));
            CardPool side = deck.get(DeckSection.Sideboard);
            if (side != null && !side.isEmpty()) {
                sb.append("\nSideboard\n");
                appendArenaSection(sb, side);
            }
            CardPool cmdr = deck.get(DeckSection.Commander);
            if (cmdr != null && !cmdr.isEmpty()) {
                sb.append("\nCommander\n");
                appendArenaSection(sb, cmdr);
            }
            return sb.toString().trim();
        }
        private static void appendArenaSection(StringBuilder sb, CardPool pool) {
            for (Map.Entry<PaperCard, Integer> e : pool) {
                PaperCard pc = e.getKey();
                sb.append(e.getValue()).append(" ").append(pc.getName())
                  .append(" (").append(pc.getEdition()).append(") ")
                  .append(pc.getCollectorNumber()).append("\n");
            }
        }

        // Forge .dck: full metadata format using DeckSerializer pattern
        private static String formatForge(Deck deck) {
            StringBuilder sb = new StringBuilder();
            sb.append("[metadata]\n");
            sb.append("Name=").append(deck.getName()).append("\n");
            if (deck.getComment() != null && !deck.getComment().isEmpty()) {
                sb.append("Comment=").append(deck.getComment()).append("\n");
            }
            sb.append("[Main]\n");
            appendForgeSection(sb, deck.getOrCreate(DeckSection.Main));
            CardPool side = deck.get(DeckSection.Sideboard);
            if (side != null && !side.isEmpty()) {
                sb.append("[Sideboard]\n");
                appendForgeSection(sb, side);
            }
            CardPool cmdr = deck.get(DeckSection.Commander);
            if (cmdr != null && !cmdr.isEmpty()) {
                sb.append("[Commander]\n");
                appendForgeSection(sb, cmdr);
            }
            return sb.toString().trim();
        }
        private static void appendForgeSection(StringBuilder sb, CardPool pool) {
            // Forge format: "count CardName|SetCode"
            sb.append(pool.toCardList("\n")).append("\n");
        }

        // Replicate DeckHandler.findDeckFile pattern (private in DeckHandler)
        private static File findDeckFile(String name) {
            File decksDir = new File(ForgeConstants.DECK_CONSTRUCTED_DIR);
            if (!decksDir.exists()) return null;
            File direct = new File(decksDir, name + ".dck");
            if (direct.exists()) return direct;
            return scanForDeck(decksDir, name);
        }
        private static File scanForDeck(File dir, String name) {
            File[] files = dir.listFiles();
            if (files == null) return null;
            for (File file : files) {
                if (file.isDirectory()) {
                    File found = scanForDeck(file, name);
                    if (found != null) return found;
                } else if (file.getName().endsWith(".dck")) {
                    Deck deck = DeckSerializer.fromFile(file);
                    if (deck != null && name.equals(deck.getName())) return file;
                }
            }
            return null;
        }
    }
    ```
