**2. Update useGameWebSocket to accept gameConfig:**

```typescript
export function useGameWebSocket(
  gameId: string | null,
  gameConfig?: { deckName: string; aiDeckName: string | null; format: string; aiDifficulty: string }
) {
  const wsRef = useRef<GameWebSocket | null>(null)

  useEffect(() => {
    if (!gameId) return
    if (wsRef.current !== null) return

    const ws = new GameWebSocket(gameId)
    wsRef.current = ws
    ws.connect(gameConfig)

    return () => {
      ws.disconnect()
      wsRef.current = null
      useGameStore.getState().reset()
    }
  }, [gameId]) // gameConfig intentionally NOT in deps -- only used on initial connect

  return wsRef
}
```
