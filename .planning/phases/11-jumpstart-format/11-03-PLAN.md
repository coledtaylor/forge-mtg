3. Add `loadPackByName` private static method:
   ```java
   private static Deck loadPackByName(final String name) {
       // First try: look for user deck file
       Deck userDeck = loadDeckByName(name);
       if (userDeck != null) return userDeck;

       // Second try: look for built-in Jumpstart pack
       for (SealedTemplate template : StaticData.instance().getSpecialBoosters()) {
           if (template.getName().equals(name)) {
               UnOpenedProduct product = new UnOpenedProduct(template);
               Deck deck = new Deck(name);
               deck.getMain().add(product.get());
               return deck;
           }
       }
       return null;
   }
   ```

4. Add `mergeRandomJumpstartPacks` private static method:
   ```java
   private static Deck mergeRandomJumpstartPacks() {
       List<SealedTemplate> jumpstartTemplates = new ArrayList<>();
       for (SealedTemplate template : StaticData.instance().getSpecialBoosters()) {
           String name = template.getName();
           if (name.startsWith("JMP ") || name.startsWith("J22 ")) {
               jumpstartTemplates.add(template);
           }
       }
       if (jumpstartTemplates.size() < 2) {
           Logger.warn("Not enough Jumpstart templates for AI (found {})", jumpstartTemplates.size());
           return null;
       }
       // Pick 2 random distinct templates
       int idx1 = ThreadLocalRandom.current().nextInt(jumpstartTemplates.size());
       int idx2;
       do {
           idx2 = ThreadLocalRandom.current().nextInt(jumpstartTemplates.size());
       } while (idx2 == idx1);

       SealedTemplate t1 = jumpstartTemplates.get(idx1);
       SealedTemplate t2 = jumpstartTemplates.get(idx2);
       UnOpenedProduct p1 = new UnOpenedProduct(t1);
       UnOpenedProduct p2 = new UnOpenedProduct(t2);

       Deck aiDeck = new Deck("AI Jumpstart - " + t1.getName() + " + " + t2.getName());
       aiDeck.getMain().add(p1.get());
       aiDeck.getMain().add(p2.get());
       return aiDeck;
   }
   ```
